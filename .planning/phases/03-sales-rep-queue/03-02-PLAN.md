    1. **`packages/supabase/src/realtime.ts`** — generic helper:
       ```ts
       'use client';
       import { useEffect, useRef } from 'react';
       import { createClient } from './client';

       export type BroadcastEnvelope<T = Record<string, unknown>> = {
         event: string;
         payload?: { record?: T; old_record?: T; operation?: string };
       };

       export interface UsePrivateBroadcastOptions<T> {
         topic: string;                  // e.g. `agent:${userId}`
         event?: string;                 // default '*'
         onMessage: (env: BroadcastEnvelope<T>) => void;
         onStatusChange?: (status: 'SUBSCRIBED' | 'CHANNEL_ERROR' | 'TIMED_OUT' | 'CLOSED') => void;
       }

       export function usePrivateBroadcast<T>({ topic, event = '*', onMessage, onStatusChange }: UsePrivateBroadcastOptions<T>) {
         const cb = useRef(onMessage);
         cb.current = onMessage;
         const status = useRef(onStatusChange);
         status.current = onStatusChange;

         useEffect(() => {
           const supabase = createClient();
           const channel = supabase.channel(topic, { config: { private: true } });
           channel.on('broadcast', { event }, (msg) => cb.current(msg as BroadcastEnvelope<T>));
           channel.subscribe((s) => status.current?.(s as never));
           return () => { supabase.removeChannel(channel); };
         }, [topic, event]);
       }
       ```
       Key points:
       - `config: { private: true }` is required — server-side RLS on
         `realtime.messages` (migration 00008) gates the subscribe.
       - `event: '*'` because the trigger fires on UPDATE-of-assigned_to (not
         INSERT — see broadcast test JSDoc lines 28–34).
       - useRef pattern keeps the callback stable across re-renders without
         re-subscribing.

    2. **`apps/web/app/(sales-rep)/_components/use-agent-broadcast.ts`** — typed
       wrapper specific to this surface:
       ```ts
       'use client';
       import { usePrivateBroadcast } from '@repo/supabase/realtime';
       import type { Database } from '@repo/supabase/types';

       type LeadRow = Database['public']['Tables']['leads']['Row'];

       export function useAgentBroadcast(agentId: string, onLead: (lead: LeadRow, op: string) => void) {
         usePrivateBroadcast<LeadRow>({
           topic: `agent:${agentId}`,
           event: '*',
           onMessage: (env) => {
             const rec = env.payload?.record;
             if (rec) onLead(rec, env.payload?.operation ?? 'UPDATE');
           },
         });
       }
       ```

    Reason for separating the generic hook (in `@repo/supabase/realtime`) from
    the queue-specific wrapper (in `apps/web`): country admin (Phase 4) will
    reuse the generic hook with `topic: country:${code}`. Don't lock the
    generic hook to lead shapes.

    2. **`queue-view.tsx`** — client component, owns the realtime + tab state:
       ```tsx
       'use client';
       export function QueueView({ initialQueue, initialCompleted, initialStats, agentId }: Props) {
         const [queue, setQueue] = useState(initialQueue);
         const [completed, setCompleted] = useState(initialCompleted);
         const [stats, setStats] = useState(initialStats);
         const [tab, setTab] = useState<'to_call'|'completed'>('to_call');
         const [serviceFilter, setServiceFilter] = useState<string|null>(null);

         useAgentBroadcast(agentId, (lead, op) => {
           setQueue(prev => prependOrUpdate(prev, lead));
           // Bump to_call_count optimistically; server view stays authoritative.
           setStats(s => ({ ...s, to_call_count: s.to_call_count + (isNew(lead) ? 1 : 0) }));
         });

         const visible = tab === 'to_call' ? queue : completed;
         const filtered = serviceFilter ? visible.filter(l => l.form_slug === serviceFilter) : visible;

         return (
           <div className="space-y-6">
             <QueueStats stats={stats} />
             <QueueTabs tab={tab} onChange={setTab} stats={stats} />
             <QueueServiceFilter value={serviceFilter} onChange={setServiceFilter} />
             <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-5">
               {filtered.map(lead => <QueueCard key={lead.id} lead={lead} />)}
             </div>
             {filtered.length === 0 && <EmptyState tab={tab} hasFilter={!!serviceFilter} />}
           </div>
         );
       }
       ```
