    3. `ingest_lead(payload jsonb) RETURNS jsonb` — the atomic wrapper called by the webhook route. Behaviour:
       ```sql
       CREATE OR REPLACE FUNCTION public.ingest_lead(payload jsonb)
       RETURNS jsonb
       LANGUAGE plpgsql
       SECURITY DEFINER
       SET search_path = public
       AS $$
       DECLARE
         v_lead_id uuid;
         v_agent_id uuid;
         v_country text := payload->>'country_code';
         v_form text := payload->>'form_slug';
       BEGIN
         -- Validate references early (FK would catch it, but better error)
         IF NOT EXISTS (SELECT 1 FROM countries WHERE code = v_country) THEN
           RETURN jsonb_build_object('error','unknown_country','country_code',v_country);
         END IF;
         IF NOT EXISTS (SELECT 1 FROM forms WHERE slug = v_form) THEN
           RETURN jsonb_build_object('error','unknown_form','form_slug',v_form);
         END IF;

         -- Insert with idempotency catch
         INSERT INTO leads (
           country_code, form_slug, status,
           name, email, phone, message,
           raw_payload, source_url,
           utm_source, utm_medium, utm_campaign,
           submitted_at
         ) VALUES (
           v_country, v_form, 'new',
           payload->>'name', payload->>'email', payload->>'phone', payload->>'message',
           COALESCE(payload->'raw_payload', payload),
           payload->>'source_url',
           payload->>'utm_source', payload->>'utm_medium', payload->>'utm_campaign',
           (payload->>'submitted_at')::timestamptz
         )
         ON CONFLICT ON CONSTRAINT leads_dedupe_idx DO NOTHING
         RETURNING id INTO v_lead_id;

         IF v_lead_id IS NULL THEN
           -- Duplicate; return existing
           SELECT id INTO v_lead_id FROM leads
           WHERE form_slug = v_form
             AND COALESCE(lower(email), phone, '') = COALESCE(lower(payload->>'email'), payload->>'phone', '')
             AND date_trunc('minute', submitted_at)
                 - (extract(minute FROM submitted_at)::int % 5) * interval '1 minute'
                 = date_trunc('minute', (payload->>'submitted_at')::timestamptz)
                 - (extract(minute FROM (payload->>'submitted_at')::timestamptz)::int % 5) * interval '1 minute'
           LIMIT 1;
           RETURN jsonb_build_object('lead_id', v_lead_id, 'duplicate', true);
         END IF;

         -- Created event
         INSERT INTO lead_events (lead_id, type, payload, country_code)
           VALUES (v_lead_id, 'created', payload, v_country);

         -- Round-robin assignment (idempotent — assign_lead handles its own event log)
         v_agent_id := public.assign_lead(v_lead_id, v_country);

         RETURN jsonb_build_object(
           'lead_id', v_lead_id,
           'agent_id', v_agent_id,
           'duplicate', false
         );
       END;
       $$;

       REVOKE ALL ON FUNCTION public.ingest_lead(jsonb) FROM public, anon, authenticated;
       GRANT EXECUTE ON FUNCTION public.ingest_lead(jsonb) TO service_role;
       ```
       Note: `ON CONFLICT ON CONSTRAINT leads_dedupe_idx` works because Postgres treats unique indexes as constraints. If that syntax fails, fall back to `ON CONFLICT (form_slug, COALESCE(...), ...) DO NOTHING` matching the index expression exactly.

    2. Trigger function for agent broadcast — copy verbatim from RESEARCH Pattern 3:
       ```sql
       CREATE OR REPLACE FUNCTION public.broadcast_lead_to_agent()
       RETURNS trigger
       LANGUAGE plpgsql
       SECURITY DEFINER
       AS $$
       BEGIN
         IF NEW.assigned_to IS NOT NULL
            AND (TG_OP = 'INSERT' OR OLD.assigned_to IS DISTINCT FROM NEW.assigned_to) THEN
           PERFORM realtime.broadcast_changes(
             'agent:' || NEW.assigned_to::text,
             TG_OP, TG_OP,
             TG_TABLE_NAME, TG_TABLE_SCHEMA,
             NEW, OLD
           );
         END IF;
         RETURN NEW;
       END;
       $$;

       CREATE TRIGGER leads_broadcast_agent
       AFTER INSERT OR UPDATE OF assigned_to ON public.leads
       FOR EACH ROW EXECUTE FUNCTION public.broadcast_lead_to_agent();
       ```

    3. Trigger function for country admin broadcast (broadcasts EVERY change in country):
       ```sql
       CREATE OR REPLACE FUNCTION public.broadcast_lead_to_country()
       RETURNS trigger
       LANGUAGE plpgsql
       SECURITY DEFINER
       AS $$
       BEGIN
         PERFORM realtime.broadcast_changes(
           'country:' || NEW.country_code,
           TG_OP, TG_OP,
           TG_TABLE_NAME, TG_TABLE_SCHEMA,
           NEW, OLD
         );
         RETURN NEW;
       END;
       $$;

       CREATE TRIGGER leads_broadcast_country
       AFTER INSERT OR UPDATE ON public.leads
       FOR EACH ROW EXECUTE FUNCTION public.broadcast_lead_to_country();
       ```

    4. RLS authorization on `realtime.messages` (private channels per RESEARCH Pattern 3):
       ```sql
       CREATE POLICY "agent_own_topic" ON realtime.messages
         FOR SELECT TO authenticated
         USING (
           realtime.topic() = 'agent:' || (SELECT auth.uid())::text
           OR (SELECT auth.jwt() ->> 'user_role') = 'hq_admin'
         );

       CREATE POLICY "country_admin_country_topic" ON realtime.messages
         FOR SELECT TO authenticated
         USING (
           (SELECT auth.jwt() ->> 'user_role') = 'country_admin'

         );

       CREATE POLICY "hq_country_topic" ON realtime.messages
         FOR SELECT TO authenticated
         USING (
           (SELECT auth.jwt() ->> 'user_role') = 'hq_admin'
           AND realtime.topic() LIKE 'country:%'
         );
       ```
