         email: z.string().email().optional().nullable(),
         phone: z.string().min(5).max(40).optional().nullable(),
         message: z.string().max(5000).optional().nullable(),
         source_url: z.string().url().optional().nullable(),
         utm_source: z.string().max(120).optional().nullable(),
         utm_medium: z.string().max(120).optional().nullable(),
         utm_campaign: z.string().max(120).optional().nullable(),
       }).refine(d => d.email || d.phone, {
         message: 'Either email or phone is required',
       })

  <name>Task 2: CSV import route handler</name>
  <files>apps/web/app/api/leads/import-csv/route.ts</files>
  <action>
    Implement multipart/form-data handler:

    ```ts
    import { NextRequest } from 'next/server'
    import Papa from 'papaparse'
    import { csvRowSchema } from '@paratus/supabase/schemas/csvImport'
    import { ingestLead } from '@paratus/supabase/dal/leads'
    import { createServerClient } from '@paratus/supabase/server'

    export const runtime = 'nodejs'

    export async function POST(req: NextRequest) {
      // Auth: only authenticated country_admin or hq_admin can bulk import
      const supabase = await createServerClient()
      const { data: { user } } = await supabase.auth.getUser()
      if (!user) return Response.json({ error: 'unauthorized' }, { status: 401 })

      const { data: roleRow } = await supabase
        .from('user_roles').select('role, country_code').eq('user_id', user.id).single()
      if (!roleRow || !['country_admin', 'hq_admin'].includes(roleRow.role)) {
        return Response.json({ error: 'forbidden' }, { status: 403 })
      }

      const form = await req.formData()
      const file = form.get('file')
      if (!(file instanceof File)) return Response.json({ error: 'file missing' }, { status: 400 })

      const text = await file.text()
      const parsed = Papa.parse<Record<string, string>>(text, { header: true, skipEmptyLines: true })
      if (parsed.errors.length) return Response.json({ error: 'csv parse failed', detail: parsed.errors }, { status: 400 })

      const result = { inserted: 0, duplicates: 0, errors: [] as Array<{ row: number; message: string }> }

      // Country admins are scoped to their country — override country_code in every row
      const forcedCountry = roleRow.role === 'country_admin' ? roleRow.country_code : null

      // Chunk inserts to avoid huge transactions
      for (let i = 0; i < parsed.data.length; i++) {
        const raw = parsed.data[i]
        if (forcedCountry) raw.country_code = forcedCountry
        const validation = csvRowSchema.safeParse(raw)
        if (!validation.success) {
          result.errors.push({ row: i + 2, message: validation.error.issues.map(x => x.message).join('; ') }) // +2 for header + 1-indexed
          continue
        }
        try {
          const r = await ingestLead(validation.data)
          if ('error' in r) result.errors.push({ row: i + 2, message: r.error as string })
          else if (r.duplicate) result.duplicates++
          else result.inserted++
        } catch (e) {
          result.errors.push({ row: i + 2, message: e instanceof Error ? e.message : 'unknown' })
        }
      }

      return Response.json(result, { status: 200 })
    }
    ```

    Notes:
    - Country admins can only import for their own country (forced override).
    - HQ admins can import for any country (rows must include `country_code`).
    - Each row goes through `ingest_lead()` so idempotency, validation, and broadcast all happen automatically.
